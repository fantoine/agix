# Agix Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build Agix, a Rust CLI package manager for AI CLI tools (Claude Code, Codex) that manages skills, agents, plugins and other extensions via an Agentfile.

**Architecture:** Monolithic Rust binary with a `CliDriver` trait per supported AI CLI, a `Source` trait per package source type, and a unified Agentfile (TOML) format shared by both project manifests and package manifests. The lock file (`Agentfile.lock`) combines version pinning and installation state.

**Tech Stack:** Rust, clap 4, tokio 1, reqwest 0.12, git2 0.19, toml 0.8, serde 1, blake3 1, indicatif 0.17, thiserror 1, anyhow 1. Dev: assert_cmd, tempfile, mockito.

```rust
use std::collections::HashMap;
use serde::{Deserialize, Serialize};

#[derive(Debug, Clone, Deserialize, Serialize)]
pub struct AgixSection {
    pub cli: Vec<String>,
    pub name: Option<String>,
    pub version: Option<String>,
    pub description: Option<String>,
}

#[derive(Debug, Clone, Deserialize, Serialize)]
pub struct Dependency {
    pub source: String,
    pub version: Option<String>,
    pub exclude: Option<Vec<String>>,
}

impl<'de> serde::Deserialize<'de> for Dependency {
    // Support both string shorthand and table form
}

// Support "dep = "github:org/dep"" shorthand via custom deserialize
impl<'de> serde::Deserialize<'de> for Dependency {
    fn deserialize<D: serde::Deserializer<'de>>(d: D) -> Result<Self, D::Error> {
        use serde::de::{self, Visitor, MapAccess};

        struct DependencyVisitor;
        impl<'de> Visitor<'de> for DependencyVisitor {
            type Value = Dependency;
            fn expecting(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
                f.write_str("string or dependency table")
            }
            fn visit_str<E: de::Error>(self, v: &str) -> Result<Self::Value, E> {
                Ok(Dependency { source: v.to_owned(), version: None, exclude: None })
            }
            fn visit_map<M: MapAccess<'de>>(self, mut map: M) -> Result<Self::Value, M::Error> {
                let mut source = None;
                let mut version = None;
                let mut exclude = None;
                while let Some(key) = map.next_key::<String>()? {
                    match key.as_str() {
                        "source"  => source  = Some(map.next_value()?),
                        "version" => version = Some(map.next_value()?),
                        "exclude" => exclude = Some(map.next_value()?),
                        _ => { map.next_value::<serde::de::IgnoredAny>()?; }
                    }
                }
                Ok(Dependency {
                    source: source.ok_or_else(|| de::Error::missing_field("source"))?,
                    version,
                    exclude,
                })
            }
        }
        d.deserialize_any(DependencyVisitor)
    }
}

#[derive(Debug, Clone, Deserialize, Serialize, Default)]
pub struct Hooks {
    #[serde(rename = "post-install")]
    pub post_install: Option<String>,
    #[serde(rename = "pre-uninstall")]
    pub pre_uninstall: Option<String>,
}

#[derive(Debug, Clone, Deserialize, Serialize)]
pub struct ProjectManifest {
    pub agix: AgixSection,
    #[serde(default)]
    pub dependencies: HashMap<String, Dependency>,
    /// Keys are CLI names, e.g. "claude-code"
    #[serde(flatten)]
    pub cli_dependencies: HashMap<String, CliSection>,
}

#[derive(Debug, Clone, Deserialize, Serialize, Default)]
pub struct CliSection {
    pub dependencies: HashMap<String, Dependency>,
}

#[derive(Debug, Clone, Deserialize, Serialize)]
pub struct PackageManifest {
    pub agix: AgixSection,
    #[serde(default)]
    pub dependencies: HashMap<String, Dependency>,
    #[serde(flatten)]
    pub cli_dependencies: HashMap<String, CliSection>,
    pub hooks: Option<Hooks>,
}

impl ProjectManifest {
    pub fn from_file(path: &std::path::Path) -> crate::error::Result<Self> {
        let content = std::fs::read_to_string(path)?;
        Ok(toml::from_str(&content)?)
    }

    pub fn to_file(&self, path: &std::path::Path) -> crate::error::Result<()> {
        let content = toml::to_string_pretty(self)?;
        std::fs::write(path, content)?;
        Ok(())
    }

    pub fn empty(cli: Vec<String>) -> Self {
        ProjectManifest {
            agix: AgixSection { cli, name: None, version: None, description: None },
            dependencies: HashMap::new(),
            cli_dependencies: HashMap::new(),
        }
    }
}

impl PackageManifest {
    pub fn from_file(path: &std::path::Path) -> crate::error::Result<Option<Self>> {
        if !path.exists() { return Ok(None); }
        let content = std::fs::read_to_string(path)?;

    }
```

```rust
use crate::error::Result;

#[derive(Debug, Clone, Deserialize, Serialize)]
pub struct InstalledFile {
    pub dest: String,
}

#[derive(Debug, Clone, Deserialize, Serialize)]
pub struct LockedPackage {
    pub name: String,
    pub source: String,
    pub sha: Option<String>,
    pub content_hash: Option<String>,
    pub cli: Vec<String>,
    pub scope: String,
    #[serde(default)]
    pub files: Vec<InstalledFile>,
}

#[derive(Debug, Clone, Deserialize, Serialize, Default)]
pub struct LockFile {
    #[serde(default, rename = "package")]
    pub packages: Vec<LockedPackage>,
}

impl LockFile {
    pub fn from_file(path: &std::path::Path) -> Result<Self> {
        let content = std::fs::read_to_string(path)?;
        Ok(toml::from_str(&content)?)
    }

    pub fn from_file_or_default(path: &std::path::Path) -> Self {
        if path.exists() {
            Self::from_file(path).unwrap_or_default()
        } else {
            Self::default()
        }
    }

    pub fn to_file(&self, path: &std::path::Path) -> Result<()> {
        let content = toml::to_string_pretty(self)?;
        std::fs::write(path, content)?;
        Ok(())
    }

    pub fn find(&self, name: &str) -> Option<&LockedPackage> {
        self.packages.iter().find(|p| p.name == name)
    }

    pub fn find_mut(&mut self, name: &str) -> Option<&mut LockedPackage> {
        self.packages.iter_mut().find(|p| p.name == name)
    }

    pub fn remove(&mut self, name: &str) {
        self.packages.retain(|p| p.name != name);
    }

    pub fn upsert(&mut self, pkg: LockedPackage) {
        self.remove(&pkg.name.clone());
        self.packages.push(pkg);
    }
}
```

```rust
pub mod github;
pub mod git;
pub mod local;

use crate::error::{AgixError, Result};

#[derive(Debug, Clone)]
pub enum SourceSpec {
    GitHub { org: String, repo: String, ref_str: Option<String> },
    Git    { url: String, ref_str: Option<String> },
    Local  { path: std::path::PathBuf },
    Marketplace { cli: String, identifier: String },
}

impl SourceSpec {
    pub fn parse(s: &str) -> Result<Self> {
        if let Some(rest) = s.strip_prefix("github:") {
            let (path, ref_str) = split_ref(rest);
            let (org, repo) = path.split_once('/').ok_or_else(|| {
                AgixError::InvalidSource(format!("github source must be 'github:org/repo', got: {s}"))
            })?;
            return Ok(SourceSpec::GitHub {
                org: org.to_owned(),
                repo: repo.to_owned(),
                ref_str: ref_str.map(str::to_owned),
            });
        }
        if let Some(rest) = s.strip_prefix("git:") {
            let (url, ref_str) = split_ref(rest);
            return Ok(SourceSpec::Git {
                url: url.to_owned(),
                ref_str: ref_str.map(str::to_owned),
            });
        }
        if let Some(path) = s.strip_prefix("local:") {
            return Ok(SourceSpec::Local {
                path: std::path::PathBuf::from(path),
            });
        }
        // Marketplace: "<cli>:marketplace@<identifier>"
        if let Some(colon) = s.find(':') {
            let cli = &s[..colon];
            let rest = &s[colon + 1..];
            if rest.starts_with("marketplace@") {
                return Ok(SourceSpec::Marketplace {
                    cli: cli.to_owned(),
                    identifier: rest["marketplace@".len()..].to_owned(),
                });
            }
        }
        Err(AgixError::InvalidSource(format!("unknown source scheme: {s}")))
    }

    pub fn canonical(&self) -> String {
        match self {
            SourceSpec::GitHub { org, repo, ref_str } => {
                let base = format!("github:{org}/{repo}");
                if let Some(r) = ref_str { format!("{base}@{r}") } else { base }
            }
            SourceSpec::Git { url, ref_str } => {
                let base = format!("git:{url}");
                if let Some(r) = ref_str { format!("{base}@{r}") } else { base }
            }
            SourceSpec::Local { path } => format!("local:{}", path.display()),
            SourceSpec::Marketplace { cli, identifier } => {
                format!("{cli}:marketplace@{identifier}")
            }
        }
    }
}

fn split_ref(s: &str) -> (&str, Option<&str>) {
    match s.find('@') {
        Some(i) => (&s[..i], Some(&s[i + 1..])),
        None    => (s, None),
    }
}
```

```rust
use crate::error::Result;

pub struct FetchedLocal {
    pub content_hash: String,
    pub path: PathBuf,
}

pub struct LocalSource {
    pub path: PathBuf,
}

impl LocalSource {
    pub fn new(path: PathBuf) -> Self { Self { path } }

    pub fn fetch(&self, dest: &Path) -> Result<FetchedLocal> {
        copy_dir_all(&self.path, dest)?;
        let hash = Self::hash_dir(dest)?;
        Ok(FetchedLocal { content_hash: hash, path: dest.to_path_buf() })
    }

    pub fn hash_dir(dir: &Path) -> Result<String> {
        let mut hasher = blake3::Hasher::new();
        let mut entries: Vec<_> = walkdir::WalkDir::new(dir)
            .into_iter()
            .filter_map(|e| e.ok())
            .filter(|e| e.file_type().is_file())
            .collect();
        entries.sort_by_key(|e| e.path().to_path_buf());
        for entry in entries {
            let content = std::fs::read(entry.path())?;
            hasher.update(entry.path().to_string_lossy().as_bytes());
            hasher.update(&content);
        }
        Ok(hasher.finalize().to_hex().to_string())
    }
}

fn copy_dir_all(src: &Path, dst: &Path) -> Result<()> {
    std::fs::create_dir_all(dst)?;
    for entry in std::fs::read_dir(src)? {
        let entry = entry?;
        let ty = entry.file_type()?;
        if ty.is_dir() {
            copy_dir_all(&entry.path(), &dst.join(entry.file_name()))?;
        } else {
            std::fs::copy(entry.path(), dst.join(entry.file_name()))?;
        }
    }
    Ok(())
}
```

```rust
const GITHUB_API: &str = "https://api.github.com";

pub struct FetchedGitHub {
    pub sha: String,
    pub path: std::path::PathBuf,
}

pub struct GitHubSource {
    org: String,
    repo: String,
    ref_str: Option<String>,
    api_base: String,
}

impl GitHubSource {
    pub fn new(org: &str, repo: &str, ref_str: Option<&str>) -> Self {
        Self {
            org: org.to_owned(),
            repo: repo.to_owned(),
            ref_str: ref_str.map(str::to_owned),
            api_base: GITHUB_API.to_owned(),
        }
    }

    #[cfg(test)]
    pub fn new_with_base(org: &str, repo: &str, ref_str: Option<&str>, base: &str) -> Self {
        Self { org: org.to_owned(), repo: repo.to_owned(),
               ref_str: ref_str.map(str::to_owned), api_base: base.to_owned() }
    }

    pub fn parse_org_repo(s: &str) -> Result<(String, String)> {
        s.split_once('/').map(|(o, r)| (o.to_owned(), r.to_owned()))
            .ok_or_else(|| AgixError::InvalidSource(format!("expected org/repo, got: {s}")))
    }

    pub async fn resolve_ref(&self) -> Result<String> {
        let client = reqwest::Client::builder()
            .user_agent("agix/0.1")
            .build()?;

        let ref_str = self.ref_str.as_deref().unwrap_or("HEAD");

        // Try tag first, then branch, then treat as commit hash
        if ref_str.len() == 40 || ref_str.len() == 7 {
            // Looks like a commit hash — use directly
            return Ok(ref_str.to_owned());
        }

        // Try as tag ref
        let url = format!("{}/repos/{}/{}/git/ref/tags/{}", self.api_base, self.org, self.repo, ref_str);
        let resp = client.get(&url).send().await?;
        if resp.status().is_success() {
            let json: serde_json::Value = resp.json().await?;
            if let Some(sha) = json["object"]["sha"].as_str() {
                return Ok(sha.to_owned());
            }
        }

        // Try as branch
        let url = format!("{}/repos/{}/{}/git/ref/heads/{}", self.api_base, self.org, self.repo, ref_str);
        let resp = client.get(&url).send().await?;
        if resp.status().is_success() {
            let json: serde_json::Value = resp.json().await?;
            if let Some(sha) = json["object"]["sha"].as_str() {
                return Ok(sha.to_owned());
            }
        }

        Err(AgixError::Other(format!("could not resolve ref '{}' for {}/{}", ref_str, self.org, self.repo)))
    }

    pub async fn fetch(&self, dest: &Path) -> Result<FetchedGitHub> {
        let sha = self.resolve_ref().await?;
        let url = format!(
            "https://github.com/{}/{}/archive/{}.zip",
            self.org, self.repo, sha
        );
        let bytes = reqwest::get(&url).await?.bytes().await?;
        extract_zip(&bytes, dest)?;
        Ok(FetchedGitHub { sha, path: dest.to_path_buf() })
    }
}

fn extract_zip(bytes: &[u8], dest: &Path) -> Result<()> {
    use std::io::Cursor;
    let cursor = Cursor::new(bytes);
    let mut archive = zip::ZipArchive::new(cursor)
        .map_err(|e| AgixError::Other(e.to_string()))?;
    for i in 0..archive.len() {
        let mut file = archive.by_index(i)
            .map_err(|e| AgixError::Other(e.to_string()))?;
        // Strip the top-level directory added by GitHub
        let name = file.name().to_owned();
        let stripped = name.splitn(2, '/').nth(1).unwrap_or(&name);
        if stripped.is_empty() { continue; }
        let out = dest.join(stripped);
        if file.is_dir() {
            std::fs::create_dir_all(&out)?;
        } else {
            if let Some(parent) = out.parent() { std::fs::create_dir_all(parent)?; }
            let mut outfile = std::fs::File::create(&out)?;
            std::io::copy(&mut file, &mut outfile)?;
        }
    }
    Ok(())
}
```

```rust
    url: String,
    ref_str: Option<String>,

        let sha = head.peel_to_commit()?.id().to_string();
        Ok(FetchedGit { sha, path: dest.to_path_buf() })
    }
```

```rust
use crate::manifest::agentfile::ProjectManifest;

#[derive(Debug, Clone)]
pub struct ResolvedDep {
    pub name: String,
    pub source: String,
    pub version: Option<String>,
    pub cli: Vec<String>,
}

pub struct Resolver;

impl Resolver {
    /// Resolve all dependencies from a manifest for the given active CLIs.
    pub fn resolve(manifest: &ProjectManifest, active_clis: &[String]) -> Vec<ResolvedDep> {
        let mut resolved: Vec<ResolvedDep> = Vec::new();

        // Shared dependencies → all active CLIs, minus excluded
        for (name, dep) in &manifest.dependencies {
            let clis: Vec<String> = active_clis.iter()
                .filter(|cli| {
                    dep.exclude.as_deref()
                        .map(|ex| !ex.contains(cli.as_str()))
                        .unwrap_or(true)
                })
                .cloned()
                .collect();
            if clis.is_empty() { continue; }
            resolved.push(ResolvedDep {
                name: name.clone(),
                source: dep.source.clone(),
                version: dep.version.clone(),
                cli: clis,
            });
        }

        // CLI-specific dependencies
        for cli in active_clis {
            if let Some(section) = manifest.cli_dependencies.get(cli) {
                for (name, dep) in &section.dependencies {
                    // Merge with existing entry if same name
                    if let Some(existing) = resolved.iter_mut().find(|r| &r.name == name) {
                        if !existing.cli.contains(cli) { existing.cli.push(cli.clone()); }
                    } else {
                        resolved.push(ResolvedDep {
                            name: name.clone(),
                            source: dep.source.clone(),
                            version: dep.version.clone(),
                            cli: vec![cli.clone()],
                        });
                    }
                }
            }
        }

        resolved
    }
}
```

```rust
use std::path::Path;
use crate::core::lock::{LockFile, LockedPackage};
use crate::core::resolver::Resolver;
use crate::drivers::{driver_for, FetchedPackage, Scope};
use crate::error::{AgixError, Result};
use crate::manifest::agentfile::ProjectManifest;
use crate::sources::{SourceSpec, github::GitHubSource, git::GitSource, local::LocalSource};

pub struct Installer;

impl Installer {
    pub async fn install_manifest(
        manifest: &ProjectManifest,
        lock_path: &Path,
        scope_str: &str,
        base: &Path,
    ) -> Result<()> {
        let scope = if scope_str == "global" { Scope::Global } else { Scope::Local };
        let active_clis = &manifest.agix.cli;
        let deps = Resolver::resolve(manifest, active_clis);
        let mut lock = LockFile::from_file_or_default(lock_path);
        let tmp = tempfile::tempdir()?;

        for dep in &deps {
            let fetch_dir = tmp.path().join(&dep.name);
            std::fs::create_dir_all(&fetch_dir)?;

            let spec = SourceSpec::parse(&dep.source)?;
            let fetched = match &spec {
                SourceSpec::Local { path } => {
                    let src = LocalSource::new(path.clone());
                    let f = src.fetch(&fetch_dir)?;
                    FetchedPackage { path: f.path, sha: None, content_hash: Some(f.content_hash) }
                }
                SourceSpec::GitHub { org, repo, ref_str } => {
                    let src = GitHubSource::new(org, repo, ref_str.as_deref());
                    let f = src.fetch(&fetch_dir).await?;
                    FetchedPackage { path: f.path, sha: Some(f.sha), content_hash: None }
                }
                SourceSpec::Git { url, ref_str } => {
                    let src = GitSource::new(url, ref_str.as_deref());
                    let f = src.fetch(&fetch_dir)?;
                    FetchedPackage { path: f.path, sha: Some(f.sha), content_hash: None }
                }
                SourceSpec::Marketplace { cli, identifier } => {
                    let driver = driver_for(cli).ok_or_else(|| AgixError::Other(format!("no driver for {cli}")))?;
                    let files = driver.install_from_marketplace(identifier, &scope)?;
                    lock.upsert(LockedPackage {
                        name: dep.name.clone(),
                        source: dep.source.clone(),
                        sha: None, content_hash: None,
                        cli: dep.cli.clone(),
                        scope: scope_str.to_owned(),
                        files,
                    });
                    lock.to_file(lock_path)?;
                    continue;
                }
            };

            let mut all_files = Vec::new();
            for cli_name in &dep.cli {
                let driver = driver_for(cli_name)
                    .ok_or_else(|| AgixError::Other(format!("no driver for CLI: {cli_name}")))?;
                if !driver.detect() {
                    crate::output::warn(&format!("{cli_name} not detected — skipping {}", dep.name));
                    continue;
                }
                let files = driver.install(&dep.name, &fetched, &scope)?;
                all_files.extend(files);
            }

            lock.upsert(LockedPackage {
                name: dep.name.clone(),
                source: dep.source.clone(),
                sha: fetched.sha,
                content_hash: fetched.content_hash,
                cli: dep.cli.clone(),
                scope: scope_str.to_owned(),
                files: all_files,
            });
            lock.to_file(lock_path)?;
        }
        Ok(())
    }

    pub fn uninstall(name: &str, lock_path: &Path) -> Result<()> {
        let mut lock = LockFile::from_file_or_default(lock_path);
        if let Some(pkg) = lock.find(name) {
            let cli_names = pkg.cli.clone();
            let files = pkg.files.clone();
            for cli_name in &cli_names {
                if let Some(driver) = driver_for(cli_name) {
                    driver.uninstall(&files)?;
                }
            }
        }
        lock.remove(name);
        lock.to_file(lock_path)?;
        Ok(())
    }
}
```

```rust
pub async fn run(source: String, global: bool, cli_filter: Option<String>, version: Option<String>) -> Result<()> {
    let (agentfile_path, lock_path, scope) = paths(global)?;

    if let Some(ref cli) = cli_filter {
        manifest.cli_dependencies

            .insert(name.clone(), dep);

    Installer::install_manifest(&manifest, &lock_path, scope, &std::env::current_dir()?)

}
```

```rust
use anyhow::{Context, Result};
use crate::core::installer::Installer;
use crate::core::lock::LockFile;
use crate::manifest::agentfile::ProjectManifest;

pub async fn run(name: Option<String>, global: bool) -> Result<()> {
    let (agentfile_path, lock_path, scope) = paths(global)?;

    if !agentfile_path.exists() {
        anyhow::bail!("No Agentfile found.");
    }

    let manifest = ProjectManifest::from_file(&agentfile_path)
        .context("Failed to parse Agentfile")?;

    if let Some(pkg_name) = name {
        // Uninstall specific package, then re-install
        Installer::uninstall(&pkg_name, &lock_path)?;
    } else {
        // Uninstall all, then re-install everything
        let lock = LockFile::from_file_or_default(&lock_path);
        for pkg in &lock.packages {
            Installer::uninstall(&pkg.name, &lock_path)?;
        }
    }

    Installer::install_manifest(&manifest, &lock_path, scope, &std::env::current_dir()?)
        .await
        .context("Re-installation failed")?;

    crate::output::success("Updated");
    Ok(())
}

fn paths(global: bool) -> Result<(std::path::PathBuf, std::path::PathBuf, &'static str)> {
    if global {
        let home = dirs::home_dir().ok_or_else(|| anyhow::anyhow!("no home dir"))?;
        let dir = home.join(".agix");
        Ok((dir.join("Agentfile"), dir.join("Agentfile.lock"), "global"))
    } else {
        let dir = std::env::current_dir()?;
        Ok((dir.join("Agentfile"), dir.join("Agentfile.lock"), "local"))
    }
}
```

```rust
use anyhow::Result;
use crate::core::lock::LockFile;
use crate::drivers::driver_for;
use crate::manifest::agentfile::ProjectManifest;

pub async fn run() -> Result<()> {
    let dir = std::env::current_dir()?;
    let agentfile_path = dir.join("Agentfile");
    let lock_path = dir.join("Agentfile.lock");

    println!("Agix Doctor\n");

    if !agentfile_path.exists() {
        crate::output::warn("No Agentfile in current directory");
        return Ok(());
    }

    let manifest = ProjectManifest::from_file(&agentfile_path)?;
    for cli in &manifest.agix.cli {
        if let Some(driver) = driver_for(cli) {
            if driver.detect() {
                crate::output::success(&format!("{cli} detected"));
            } else {
                crate::output::warn(&format!("{cli} declared but not detected"));
            }
        } else {
            crate::output::warn(&format!("{cli} — no driver available in this version of agix"));
        }
    }

    let lock = LockFile::from_file_or_default(&lock_path);
    let mut missing = 0usize;
    for pkg in &lock.packages {
        for file in &pkg.files {
            let p = std::path::Path::new(&file.dest);
            if !p.exists() {
                crate::output::warn(&format!("missing: {} (from {})", file.dest, pkg.name));
                missing += 1;
            }
        }
    }
    if missing == 0 {
        crate::output::success("All installed files present");
    } else {
        println!("\n  Run `agix install` to restore missing files.");
    }
    Ok(())
}
```

```rust
use anyhow::{Context, Result};
use std::path::Path;

pub async fn run(global: bool, all: bool, output: Option<String>) -> Result<()> {
    let output_path = output.as_deref().unwrap_or("agix-export.zip");
    let file = std::fs::File::create(output_path)?;
    let mut zip = zip::ZipWriter::new(file);
    let options: zip::write::FileOptions<()> = zip::write::FileOptions::default();

    if all || !global {
        let dir = std::env::current_dir()?;
        add_file_to_zip(&mut zip, &dir.join("Agentfile"), "Agentfile", &options)
            .context("Agentfile not found")?;
        if dir.join("Agentfile.lock").exists() {
            add_file_to_zip(&mut zip, &dir.join("Agentfile.lock"), "Agentfile.lock", &options)?;
        }
        add_local_sources_to_zip(&mut zip, &dir, &options)?;
    }

    if all || global {
        if let Some(home) = dirs::home_dir() {
            let agix_dir = home.join(".agix");
            if agix_dir.join("Agentfile").exists() {
                add_file_to_zip(&mut zip, &agix_dir.join("Agentfile"), "global/Agentfile", &options)?;
            }
            if agix_dir.join("Agentfile.lock").exists() {
                add_file_to_zip(&mut zip, &agix_dir.join("Agentfile.lock"), "global/Agentfile.lock", &options)?;
            }
        }
    }

    zip.finish()?;
    crate::output::success(&format!("Exported to {output_path}"));
    Ok(())
}

fn add_file_to_zip<W: std::io::Write + std::io::Seek>(
    zip: &mut zip::ZipWriter<W>,
    path: &Path,
    name: &str,
    options: &zip::write::FileOptions<()>,
) -> Result<()> {
    zip.start_file(name, *options)?;
    let content = std::fs::read(path)?;
    std::io::Write::write_all(zip, &content)?;
    Ok(())
}

fn add_local_sources_to_zip<W: std::io::Write + std::io::Seek>(
    zip: &mut zip::ZipWriter<W>,
    project_dir: &Path,
    options: &zip::write::FileOptions<()>,
) -> Result<()> {
    use crate::core::lock::LockFile;
    use crate::sources::SourceSpec;

    let lock = LockFile::from_file_or_default(&project_dir.join("Agentfile.lock"));
    for pkg in &lock.packages {
        if let Ok(SourceSpec::Local { path }) = SourceSpec::parse(&pkg.source) {
            let resolved = if path.is_absolute() { path } else { project_dir.join(&path) };
            if resolved.exists() {
                for entry in walkdir::WalkDir::new(&resolved) {
                    let entry = entry?;
                    if entry.file_type().is_file() {
                        let rel = entry.path().strip_prefix(&resolved).unwrap();
                        let zip_name = format!("local-sources/{}/{}", pkg.name, rel.display());
                        zip.start_file(&zip_name, *options)?;
                        let content = std::fs::read(entry.path())?;
                        std::io::Write::write_all(zip, &content)?;
                    }
                }
            }
        }
    }
    Ok(())
}
```
